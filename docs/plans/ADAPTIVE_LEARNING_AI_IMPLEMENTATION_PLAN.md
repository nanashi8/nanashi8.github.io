        hidden_size: int = 64,     # LSTMéš ã‚Œå±¤ã®ã‚µã‚¤ã‚º
        num_layers: int = 2,       # LSTMã®å±¤æ•°
        num_classes: int = 5,      # å‡ºåŠ›ã‚¯ãƒ©ã‚¹æ•°ï¼ˆ5ãƒ•ã‚§ãƒ¼ã‚ºï¼‰
        dropout: float = 0.3
    ):
        super().__init__()
        
        # ç‰¹å¾´é‡æ­£è¦åŒ–
        self.feature_norm = nn.BatchNorm1d(input_size)
        
        # LSTMå±¤ï¼ˆæ™‚ç³»åˆ—ãƒ‘ã‚¿ãƒ¼ãƒ³å­¦ç¿’ï¼‰
        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            dropout=dropout,
            batch_first=True,
            bidirectional=True  # åŒæ–¹å‘LSTM
        )
        
        # Attentionæ©Ÿæ§‹ï¼ˆé‡è¦ãªæ™‚ç‚¹ã«æ³¨ç›®ï¼‰
        self.attention = nn.MultiheadAttention(
            embed_dim=hidden_size * 2,  # åŒæ–¹å‘ãªã®ã§2å€
            num_heads=4,
            dropout=dropout
        )
        
        # å…¨çµåˆå±¤
        self.fc = nn.Sequential(
            nn.Linear(hidden_size * 2, 128),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(64, num_classes)
        )
        
    def forward(self, x):
        # x: (batch_size, seq_len, input_size)
        
        # ç‰¹å¾´é‡æ­£è¦åŒ–
        batch_size, seq_len, _ = x.shape
        x = x.view(-1, x.size(2))
        x = self.feature_norm(x)
        x = x.view(batch_size, seq_len, -1)
        
        # LSTM
        lstm_out, (hidden, cell) = self.lstm(x)
        # lstm_out: (batch_size, seq_len, hidden_size*2)
        
        # Attention
        attn_out, attn_weights = self.attention(
            lstm_out, lstm_out, lstm_out
        )
        
        # æœ€å¾Œã®æ™‚ç‚¹ã®å‡ºåŠ›ã‚’ä½¿ç”¨
        final_out = attn_out[:, -1, :]
        
        # å…¨çµåˆå±¤
        output = self.fc(final_out)
        
        return output, attn_weights

def train_model(
    model: nn.Module,
    train_loader: DataLoader,
    val_loader: DataLoader,
    epochs: int = 50,
    device: str = 'cuda'
):
    model = model.to(device)
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=0.001)
    scheduler = optim.lr_scheduler.ReduceLROnPlateau(
        optimizer, mode='min', patience=5
    )
    
    best_val_accuracy = 0.0
    
    for epoch in range(epochs):
        # è¨“ç·´ãƒ•ã‚§ãƒ¼ã‚º
        model.train()
        train_loss = 0.0
        train_correct = 0
        train_total = 0
        
        for features, labels in train_loader:
            features = features.to(device)
            labels = labels.to(device)
            
            optimizer.zero_grad()
            outputs, _ = model(features)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            
            train_loss += loss.item()
            _, predicted = torch.max(outputs, 1)
            train_total += labels.size(0)
            train_correct += (predicted == labels).sum().item()
        
        train_accuracy = train_correct / train_total
        
        # æ¤œè¨¼ãƒ•ã‚§ãƒ¼ã‚º
        model.eval()
        val_loss = 0.0
        val_correct = 0
        val_total = 0
        
        with torch.no_grad():
            for features, labels in val_loader:
                features = features.to(device)
                labels = labels.to(device)
                
                outputs, _ = model(features)
                loss = criterion(outputs, labels)
                
                val_loss += loss.item()
                _, predicted = torch.max(outputs, 1)
                val_total += labels.size(0)
                val_correct += (predicted == labels).sum().item()
        
        val_accuracy = val_correct / val_total
        
        # å­¦ç¿’ç‡ã®èª¿æ•´
        scheduler.step(val_loss)
        
        print(f"Epoch {epoch+1}/{epochs}")
        print(f"  Train Loss: {train_loss:.4f}, Train Acc: {train_accuracy:.4f}")
        print(f"  Val Loss: {val_loss:.4f}, Val Acc: {val_accuracy:.4f}")
        
        # ãƒ™ã‚¹ãƒˆãƒ¢ãƒ‡ãƒ«ã®ä¿å­˜
        if val_accuracy > best_val_accuracy:
            best_val_accuracy = val_accuracy
            torch.save(model.state_dict(), 'best_model.pth')
    
    print(f"\nBest Validation Accuracy: {best_val_accuracy:.4f}")
    
    return model

class ONNXMemoryPredictor {

  private modelLoaded: boolean = false;

  private extractFeatures(word: string): number[] {

    return [
      this.normalizeTime(status.timeSinceLastReview),
      this.normalizeCount(status.reviewCount),
      status.correctRate,
      this.normalizeTime(status.averageResponseTime),
      this.normalizeCount(status.consecutiveCorrect),
      this.normalizeCount(status.consecutiveWrong),
      status.difficulty,
      this.normalizeLength(word.length),
      this.normalizeTime(this.getTimeOfDay()),
      this.getSessionProgress(),
      this.normalizeCount(status.daysSinceFirstSeen),
      this.normalizeTime(status.totalStudyTime),
      this.normalizeCount(status.intradayReviewCount),
      this.normalizeCount(status.spacingReviewCount),
      this.normalizeTime(status.lastInterval),
      status.forgettingRate,
      status.learningSpeed,
      this.phaseToNumber(status.previousPhase),
      this.normalizeCount(status.phaseTransitionCount),
      status.errorPattern
    ];
  }

  private prepareSequence(word: string): number[][] {
    const history = getLearningHistory(word);
    const sequence: number[][] = [];
    
    // éå»10å›åˆ†ã®ç‰¹å¾´é‡ã‚’å–å¾—
    for (let i = Math.max(0, history.length - 10); i < history.length; i++) {
      const features = this.extractFeaturesFromHistory(history[i]);
      sequence.push(features);
    }
    
    // 10å›æœªæº€ã®å ´åˆã¯ã‚¼ãƒ­ãƒ‘ãƒ‡ã‚£ãƒ³ã‚°
    while (sequence.length < 10) {
      sequence.unshift(new Array(20).fill(0));
    }
    
    return sequence;
  }
  
  // ãƒ•ã‚§ãƒ¼ã‚ºäºˆæ¸¬ï¼ˆ<50msï¼‰
  async predict(word: string): Promise<PredictionResult> {
    if (!this.session) {
      throw new Error('Model not loaded. Call loadModel() first.');
    }
    
    const startTime = performance.now();
    
    // æ™‚ç³»åˆ—ãƒ‡ãƒ¼ã‚¿ã®æº–å‚™
    const sequence = this.prepareSequence(word);
    
    // ONNX Tensorã«å¤‰æ›
    const inputTensor = new ort.Tensor(
      'float32',
      Float32Array.from(sequence.flat()),
      [1, 10, 20]  // [batch_size, seq_len, input_size]
    );
    
    // æ¨è«–å®Ÿè¡Œ
    const outputs = await this.session.run({ input: inputTensor });
    
    // çµæœã®è§£æ
    const outputData = outputs.output.data as Float32Array;
    const attentionData = outputs.attention_weights.data as Float32Array;
    
    // Softmaxã§ç¢ºç‡ã«å¤‰æ›
    const probabilities = this.softmax(Array.from(outputData));
    
    // æœ€ã‚‚ç¢ºç‡ã®é«˜ã„ãƒ•ã‚§ãƒ¼ã‚ºã‚’é¸æŠ
    const maxIndex = probabilities.indexOf(Math.max(...probabilities));
    const phase = this.indexToPhase(maxIndex);
    const confidence = probabilities[maxIndex];
    
    const inferenceTime = performance.now() - startTime;
    console.log(`Inference time: ${inferenceTime.toFixed(1)}ms`);
    
    return {
      phase,
      confidence,
      attentionWeights: Array.from(attentionData).slice(0, 10)
    };
  }
  
  // Softmaxé–¢æ•°
  private softmax(arr: number[]): number[] {
    const maxVal = Math.max(...arr);
    const expArr = arr.map(x => Math.exp(x - maxVal));
    const sumExp = expArr.reduce((a, b) => a + b, 0);
    return expArr.map(x => x / sumExp);
  }
  
  // ãƒ•ã‚§ãƒ¼ã‚ºã®ã‚¤ãƒ³ãƒ‡ãƒƒã‚¯ã‚¹å¤‰æ›
  private indexToPhase(index: number): LearningPhase {
    const phases = [
      LearningPhase.ENCODING,
      LearningPhase.INITIAL_CONSOLIDATION,
      LearningPhase.INTRADAY_REVIEW,
      LearningPhase.SHORT_TERM,
      LearningPhase.LONG_TERM
    ];
    return phases[index];
  }
}
